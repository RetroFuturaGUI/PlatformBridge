cmake_minimum_required(VERSION 3.16)
project(PlatformBridge VERSION 0.1.0 LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

message(STATUS "Host OS: ${CMAKE_SYSTEM_NAME}")
message(STATUS "Host Processor: ${CMAKE_SYSTEM_PROCESSOR}")

if(CMAKE_SYSTEM_NAME STREQUAL "Windows")
    message(STATUS "Configuring for Windows platform")
    
    add_library("PlatformBridge" STATIC
        "${CMAKE_CURRENT_SOURCE_DIR}/source/Windows/Monitors.cpp"
    )

    target_include_directories("PlatformBridge" PUBLIC
        "${CMAKE_CURRENT_SOURCE_DIR}/source"
        "${CMAKE_CURRENT_SOURCE_DIR}/source/Windows"
    )

    # Make headers available to parent projects
    target_include_directories("PlatformBridge" PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/source>
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/source/Windows>
        $<INSTALL_INTERFACE:include> # For find_package
    )

    # Alias for linking
    add_library(PlatformBridge::PlatformBridge ALIAS "PlatformBridge")

else()
    message(FATAL_ERROR "Unsupported platform: ${CMAKE_SYSTEM_NAME}")
endif()

# Export targets for install and find_package
install(TARGETS "PlatformBridge"
    EXPORT PlatformBridgeTargets
    ARCHIVE DESTINATION lib
    LIBRARY DESTINATION lib
    PUBLIC_HEADER DESTINATION include/PlatformBridge
    FILE_SET HEADERS  # Explicitly declare public headers
)

# Generate and install PlatformBridgeConfig.cmake for find_package
install(
    EXPORT PlatformBridgeTargets
    FILE PlatformBridgeConfig.cmake
    NAMESPACE PlatformBridge::
    DESTINATION lib/cmake/PlatformBridge
)

# Include all public headers in the FILE_SET
target_sources("PlatformBridge" PUBLIC
    FILE "${CMAKE_CURRENT_SOURCE_DIR}/source/PlatformBridge.hpp"
)

enable_testing()
